string length()

returns the length of a string

 let arr = "i am venkatesh";
console.log(arr.length); 		output : 14
string slice()

returns the extracted part in a new string

 let arr = "i am venkatesh";
console.log(arr.slice(6,14)); 		output : venkatesh
string substring()

(start, end)

 let arr = 'i am venkatesh';
let array = arr.substring(5,11);
console.log(array); 		output : venkat
string substr()

(start, length)

 let arr = 'i am venkatesh';
let array = arr.substr(5,6);
console.log(array); 		output : venkat
string replace()

replaces a specified value with another value in a string

 let arr = "i am anand";
let array = arr.replace("anand","venkatesh");
console.log(array); 		output : venkatesh
string replaceAll()

If the parameter is a regular expression, the global flag (g) must be set, otherwise a TypeError is thrown.

 let arr = "i am venkatesh";
abc = arr.replaceAll("i am venkatesh","I Am Venkatesh");
console.log(abc); 		output : I Am Venkatesh

joins two or more strings

let arr2 = arr.concat(" ",arr1);

string trim()

removes whitespace from both sides of a string

 let arr = " i am venkatesh ";
console.log(arr.trim()); 		output : i am venkatesh

console.log(array); 		output : aadhi venkat
string trimEnd()

removes the whitespace only from the End of a string

 let arr = " aadhi venkat ";
let array = arr.trimEnd();
console.log(array); 		output : aadhi venkat

string charAt()

returns the character at a specified index (position) in a string

 let arr = "venkatesh";
console.log(arr.charAt(8)); 		output : h
string charCodeAt()

returns the unicode of the character at a specified index in a string

 let arr = "venkatesh";
console.log(arr.charCodeAt(8)); 		output : 104

string split

A string can be converted to an array with the split() method

 let arr = "v,e,n,k,a,t,e,s,h";
console.log(arr.split(",")); 		output : [ 'v', 'e', 'n', 'k', 'a', 't', 'e', 's', 'h' ]
